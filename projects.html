Project 1: Change Text Color

Click the button to change my color!

HTML


<p class="color-text">
   Click the button to change my color!
</p>
<button class="color-btn">
   Change Color
</button>

CSS


.color-text {
   margin: 1rem 0;
   font-size: 1.2rem;
}

button {
   padding: 0.5rem 1rem;
   background: #2563eb;
   color: white;
   border: none;
   border-radius: 4px;
   cursor: pointer;
}

JavaScript


let colors = ['red', 'blue', 'orange', 
   'green', 'yellow', 'pink', 'purple'];

document.querySelector('.color-btn')
   .addEventListener('click', () => {
       let colorText = document
           .querySelector('.color-text');
       let randomColor = colors[Math.floor(
           Math.random() * colors.length)];
       colorText.style.color = randomColor;
   });

Project 2: Image Slider

HTML


<div class="slider-container">
   <img src="" height="180" width="300">
   <div class="slider-buttons">
       <button class="prev-btn">
           Previous
       </button>
       <button class="next-btn">
           Next
       </button>
   </div>
</div>

CSS


.slider-container img {
   height: 180px;
   width: 300px;
   display: block;
   margin: 1rem auto;
}

.slider-buttons {
   text-align: center;
}

JavaScript


let Images = [
   'image1.jpg',
   'image2.jpg',
   'image3.jpg'
];

let currentIndex = 0;
let imageSlider = document.querySelector(
   '.slider-container img');
imageSlider.src = Images[currentIndex];

document.querySelector('.next-btn')
   .addEventListener('click', () => {
       currentIndex = 
           (currentIndex + 1) % Images.length;
       imageSlider.src = Images[currentIndex];
   });

document.querySelector('.prev-btn')
   .addEventListener('click', () => {
       currentIndex = (currentIndex - 1 + 
           Images.length) % Images.length;
       imageSlider.src = Images[currentIndex];
   });
